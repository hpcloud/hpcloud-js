module.exports = Subdir;

/**
 * Representation of object seperators in paths as subdirectories.
 * 
 * @class Subdir
 * @constructor
 */
function Subdir(path, delimiter) {
  this._path = path;
  this._delimiter = delimiter;
}

/**
 * This is for a degree of transparency between Subdir and ObjectInfo objects.
 *
 * @method name
 */
Subdir.prototype.name = function () {
  return this._path;
};

/**
 * Alias of name().
 *
 * @method path
 */
Subdir.prototype.path = Subdir.prototype.name;

/**
 * The delimiter used as the directory seperator.
 *
 * @return {String} The delimiter used as the directory seperator.
 */
Subdir.prototype.delimiter = function () {
  return this._delimiter;
};
